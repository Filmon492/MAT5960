\chapter{Asymptotic Compatibility (AC) of the First-Order Numerical Methods}\label{chap:chapter_4}
A numerical scheme of the form \eqref{eq:generic_H_N} is called asymptotically compatible if its numerical solution converges to the unique entropy solution of the local model \eqref{eq:local model} as both \(\Delta x \to 0\) and \(\epsilon \to 0\) simultaneously; see \cite{tianAsymptoticallyCompatibleSchemes2014}. In this chapter, we present the stability analysis of the numerical method, which provides the main ingredients needed to establish asymptotic compatibility.

\section{Stability Analysis of the Numerical Methods}
We consider the stability analysis of the numerical method given by \eqref{eq:generic_H_N}. For the local model, if we have a conservative, consistent and monotone scheme, then the scheme converges to the correct solution; see \cite[Chapter~4]{Ulrik2015}. However, for the nonlocal model it is difficult to obtain a monotone scheme. Nevertheless, under certain assumptions on the initial data, Huang and Du; see \cite{huangAsymptoticCompatibilityClass2024}, have deduced a monotone numerical scheme. Further, they have proved the asymptotic compatibility of a class of finite-volume schemes for the nonlocal traffic-flow model. To make the analysis easier to follow, we restrict ourselves to the Godunov-type flux and present the corresponding stability analysis in this chapter.

We start making the necessary assumptions on the nonlocal kernel $\omega^{\epsilon}(y)$, initial data $\rho_0$, numerical quadrature weights $w_k$ and CFL condition $\lambda$.
\begin{assumption}\label{assm:assm_1}
    \leavevmode
    The nonlocal kernel $\omega^{\epsilon}: \mathbb{R} \rightarrow \mathbb{R}$ satisfying $\omega^{\epsilon} \geq 0$ and $\int_{0}^{\epsilon} \omega^{\epsilon}(y)  \, dy = 1$ is given by
    \[
        \omega^{\epsilon}(y) =
        \begin{cases}
            \frac{1}{\epsilon} \omega\!\left(\frac{y}{\epsilon}\right), & y \in [0,\epsilon] \\
            0                                                           & \text{otherwise}
        \end{cases}
    \]
    where $\omega \in C_{c}^1([0,1])$ is a nonnegative, strictly decreasing probability density satisfying
    \[
        \int_0^1 \omega(z)\,dz = 1.
    \]
\end{assumption}
\begin{assumption}\label{assm:assm_2}
    The initial data $\rho_0$ belongs to $L^\infty(\mathbb{R})$, has bounded total variation, and satisfies
    \[
        0\leq \rho_{\min} \leq \rho_0(x) \leq \rho_{\max} \le 1,
        \qquad \text{for all } x \in \mathbb{R}.
    \]
\end{assumption}

\begin{assumption}\label{assm:assm_3}
    The numerical quadrature weights $\{w_k\}_{0 \le k \le m-1}$ satisfy
    \begin{enumerate}
        \item
              \[
                  \omega^{\epsilon}(k\Delta x) \Delta x \ge w_k \ge \omega^{\epsilon}((k+1)\Delta x) \Delta x \qquad \text{for } k= 0,1, \dots, m-1
              \]
        \item
              \[
                  w_{k-1} - w_{k} \ge c\,m^{-2}, \qquad \text{for } k= 1,2, \dots, m,
              \]
              with the convention $w_m=0$, and for some constant $c>0$ depending only on the kernel $\omega$.
        \item normalization condition
              \[
                  \sum_{k=0}^{m-1} w_k = 1.
              \]
    \end{enumerate}
\end{assumption}

\begin{assumption}\label{assm:assm_4}
    The CFL condition $\lambda$  satisfies
    \[ \lambda < \frac{1}{2}.\]
\end{assumption}

\subsection{Maximum Principle}\label{subsec:mono}

It is easy to establish the maximum principle if we have monotone numerical schemes. So, we check first if the update function $\mathcal{H}$ with corresponding Godunov numerical flux $g$ is monotone. To do that, suppose Assumptions~\ref{assm:assm_1}--\ref{assm:assm_4} holds. In addition, suppose that
\[0 \leq \rho_{\min} \leq  \rho_{j+k}^{n} \leq \rho_{\max} \leq 1 \qquad \text{ for } k = -1, 0,1,\dots,m.\]
Then we have
\begin{equation}\label{eq:q_j_equality}
    \begin{split}
        q_{j}^{n}  = \sum_{k=0}^{m-1} w_k \rho_{j+k}^{n} \leq \sum_{k=0}^{m-1} w_k \rho_{\max}= \rho_{\max},
    \end{split}
\end{equation}
where we have used $\sum_{k=0}^{m-1} w_k = 1$ by the normalization condition. By a similar argument $q_{j}^{n}$ is lower bounded by $\rho_{\min}$. Consequently, we have
\begin{equation}\label{eq:q_bound}
    0 \leq q_{j}^{n} \leq 1 \text{ for all } j\in \mathbb{Z}, n \in \mathbb{N}.
\end{equation}

The update function $\mathcal{H}$ with respect to the Godunov numerical flux
\[
    g(\rho_L,\rho_R,q_L,q_R)= \rho_L (1-q_R)
\]
is given by
\begin{equation}\label{eq:mono_H}
    \mathcal{H}( \rho_{j-1}^n, \rho_{j}^n,  \rho_{j+1}^n,\dots,  \rho_{j+m}^n)
    =
    \rho_j^n
    +
    \lambda
    \left[
        \rho_{j-1}^n(1-q_j^n)
        -
        \rho_j^n(1-q_{j+1}^n)
        \right].
\end{equation}
Taking the partial derivatives gives
\begin{align*}
    \frac{\partial \mathcal{H}}{\partial \rho_{j-1}^n} & =\lambda(1-q_j^n)                                                                                                              \\
    \frac{\partial \mathcal{H}}{\partial \rho_j^n}     & =1-\lambda\left[(1-q_{j+1}^n)+w_0\rho_{j-1}^n\right]                                                                           \\
    \frac{\partial \mathcal{H}}{\partial \rho_{j+k}^n} & = \lambda\left( w_{k-1}\rho_j^n -w_k\rho_{j-1}^n\right) \qquad \text{for } k= 1,2,\dots,m, \text{with the convention $w_m=0$}.
\end{align*}
$\mathcal{H}$ is nondecreasing with respect to $\rho_{j-1}^n$, because
\begin{equation}\label{eq:First_argument}
    \begin{split}
        \frac{\partial \mathcal{H}}{\partial \rho_{j-1}^n} & =\lambda(1-q_j^n) \geq 0 \qquad \text{ since $0 \leq q_{j}^{n} \leq 1$ due to } \eqref{eq:q_bound}.
    \end{split}
\end{equation}
Next, we check if $\mathcal{H}$ is nondecreasing with respect to $\rho_{j}^n$. We desire to have that
\[
    \frac{\partial \mathcal{H}}{\partial \rho_j^n}
    =
    1-\lambda\left[(1-q_{j+1}^n)+w_0\rho_{j-1}^n\right] \geq 0.
\]
Since $\lambda < \frac{1}{2}$ by \ref{assm:assm_4}, it suffices to show that
\[
    (1-q_{j+1}^n)+w_0\rho_{j-1}^n\leq 2.
\]
So
\begin{align*}
    (1-q_{j+1}^n)+w_0\rho_{j-1}^n & \leq (1-q_{j+1}^n)+\rho_{j-1}^n   \qquad \text{ since $0\leq w_0 \leq 1$ and $\rho_{j-1}^n \geq 0$ } \\
                                  & \leq 1+1 = 2 \qquad \text{ since $0\leq q_{j+1}^n \leq 1$ and $0 \leq \rho_{j-1}^n \leq 1$ }
\end{align*}
Therefore, $\mathcal{H}$ is nondecreasing with respect to $\rho_{j}^n$.

For $k= m$, the convention $w_m=0$ gives
\begin{align*}
    \frac{\partial \mathcal{H}}{\partial \rho_{j+m}^n} & = \lambda\left( w_{k-1}\rho_j^n -w_k\rho_{j-1}^n\right)                     \\
                                                       & =\lambda (w_{m-1}\rho_j^n)                                                  \\
                                                       & \geq \lambda cm^{-2} \rho_j^n \qquad \text{by Assumption~\ref{assm:assm_3}} \\
                                                       & \geq 0
\end{align*}
Hence, $\mathcal{H}$ is nondecreasing with respect to $\rho_{j+m}^n$
We consider now for $k= 1,2,\dots, m-1$. The derivatives are given by
\begin{align*}
    \frac{\partial \mathcal{H}}{\partial \rho_{j+k}^n} & = \lambda\left( w_{k-1}\rho_j^n -w_k\rho_{j-1}^n\right)                       \\
                                                       & =\lambda \Bigl[(w_{k-1}- w_k) \rho_j^n + w_k(\rho_j^n - \rho_{j-1}^n )\Bigr].
\end{align*}
Both the terms $\rho_j^n$ and $w_k$ are nonnegative, and by Assumption~\ref{assm:assm_3} we have
\[
    (w_{k-1}- w_k) \geq cm^{-2} > 0.
\]
We assume first that $(\rho_j^n - \rho_{j-1}^n ) \geq 0$, then the term
\[
    \lambda \Bigl[(w_{k-1}- w_k) \rho_j^n + w_k(\rho_j^n - \rho_{j-1}^n )\Bigr]
\]
is nonnegative.

We assume now that $(\rho_j^n - \rho_{j-1}^n ) < 0$, then the first term
\[
    (w_{k-1}- w_k) \rho_j^n
\]
inside the brackets is still nonnegative. However, the second term
\[
    w_k(\rho_j^n - \rho_{j-1}^n )
\]
inside the brackets is negative. Consequently, we may have $\frac{\partial \mathcal{H}}{\partial \rho_{j+k}^n} < 0$.


Thus, $\mathcal{H}$ is nondecreasing only with respect to $\rho_{j-1}^n$, $\rho_{j}^n$ and  $\rho_{j+m}^n$. So, we can not conclude that $\mathcal{H}$ is nondecreasing in each of its arguments based only on Assumptions~\ref{assm:assm_1}--\ref{assm:assm_4}. Therefore, one can not prove the maximum principle by showing $\mathcal{H}$ is monotone. Instead, we prove the maximum principle by applying the mean value theorem.


\begin{theorem}[Maximum principle]\label{the:theorem_1}
    Suppose Assumptions \ref{assm:assm_1}--\ref{assm:assm_4} holds. Let $\{\rho_j^n\}_{j \in \mathbb{Z}}^{ n \in \mathbb{N}}$ be the numerical solution computed by \eqref{eq:mono_H}. Then, the  uniform bound holds:
    \[\rho_{\min} \leq \rho_{j}^{n} \leq  \rho_{\max}  \qquad \text{for all } j \in \mathbb{Z},  n \in \mathbb{N} \]
    where
    \[
        \rho_{\min} := \inf\limits_{x\in \mathbb{R}} \rho_0 \geq 0  \quad \text{and }  \rho_{\max}:= \sup\limits_{x\in \mathbb{R}} \rho_0 \leq 1.
    \]
\end{theorem}

%------------------------------------------------------------------------------------------------
The proof is done by induction.
\begin{proof}
    For $n=0$, it is obvious that $\rho_{\min} \leq \rho_j^0 \leq \rho_{\max}$ for all $ j \in \mathbb{Z}$. Assume now that the result holds for some $n \in \mathbb{N}$, that is,
    \[
        \rho_{\min} \leq \rho_j^n \leq \rho_{\max}  \qquad \text{for all } j \in \mathbb{Z}.
    \] We show that it also holds for $n+1$. We must show that
    \[
        \rho_{\min} \leq \rho_j^{n+1} \leq \rho_{\max} \qquad \text{for all } j \in \mathbb{Z},
    \]
    or equivalently
    \[
        \rho_{\min} \leq \mathcal{H}\bigl(\rho_{j-1}^n,\rho_j^n, \rho_{j+1}^n, \dots,\rho_{j+m}^n \bigr) \leq
        \rho_{\max} \qquad \text{for all } j \in \mathbb{Z}.
    \]
    Since
    \[
        \mathcal{H}(\rho_{\min},\rho_{\min},\rho_{\min},\dots,\rho_{\min})
        =\rho_{\min},
    \]
    we write
    \[
        \mathcal{H}(\rho_{j-1}^n,\rho_j^n,\rho_{j+1}^n,\dots,\rho_{j+m}^n)-\rho_{\min}
        =
        \Delta \mathcal{H}_1+\Delta \mathcal{H}_2,
    \]
    where
    \[\Delta \mathcal{H}_1 = \mathcal{H}( \rho_{j-1}^n, \rho_{j}^n,  \rho_{j+1}^n,\dots,  \rho_{j+m}^n) - \mathcal{H}( \rho_{\min}, \rho_{\min},  \rho_{j+1}^n,\dots,  \rho_{j+m}^n)\]
    and
    \[\Delta \mathcal{H}_2 = \mathcal{H}( \rho_{\min}, \rho_{\min},  \rho_{j+1}^n,\dots,  \rho_{j+m}^n) - \mathcal{H}(\rho_{\min}, \rho_{\min}, \rho_{\min},\dots, \rho_{\min}).
    \]

    Focusing on $\Delta \mathcal{H}_1$, by the mean value theorem there exists a point $(\tilde{\rho}_{j-1}^n, \tilde{\rho}_{j}^n)$ on the line segment from $(\rho_{\min},\rho_{\min})$ to $(\rho_{j-1}^n,\rho_{j}^n)$ such that
    \begin{align*}
        \Delta \mathcal{H}_1 & =  \mathcal{H}( \rho_{j-1}^n, \rho_{j}^n,  \rho_{j+1}^n,\dots,  \rho_{j+m}^n) - \mathcal{H}( \rho_{\min}, \rho_{\min},  \rho_{j+1}^n,\dots,  \rho_{j+m}^n)                                                                                                                                                                 \\
                             & = \frac{\partial \mathcal{H}}{\partial \rho_{j-1}^n} ( \tilde{\rho}_{j-1}^n, \tilde{\rho}_{j}^n, \rho_{j+1}^n, \dots, \rho_{j+m}^n)(\rho_{j-1}^n- \rho_{\min}) + \frac{\partial \mathcal{H}}{\partial \rho_{j}^n} ( \tilde{\rho}_{j-1}^n, \tilde{\rho}_{j}^n, \rho_{j+1}^n, \dots, \rho_{j+m}^n)(\rho_{j}^n- \rho_{\min}),
    \end{align*}
    with
    \[0 \leq \rho_{\min} \leq \tilde{\rho}_{j-1}^n \leq \rho_{\max} \text{ and } 0 \leq \rho_{\min} \leq \tilde{\rho}_{j}^n \leq \rho_{\max}.\]
    We want to show that $\Delta \mathcal{H}_1 \geq 0$. It is enough to show that the two terms
    \begin{equation}\label{eq:H_1_first_term}
        \frac{\partial \mathcal{H}}{\partial \rho_{j-1}^n} ( \tilde{\rho}_{j-1}^n, \tilde{\rho}_{j}^n, \rho_{j+1}^n, \dots, \rho_{j+m}^n)(\rho_{j-1}^n- \rho_{\min})
    \end{equation}
    and
    \begin{equation}\label{eq:H_1_second_term}
        \frac{\partial \mathcal{H}}{\partial \rho_{j}^n} ( \tilde{\rho}_{j-1}^n, \tilde{\rho}_{j}^n, \rho_{j+1}^n, \dots, \rho_{j+m}^n)(\rho_{j}^n- \rho_{\min})
    \end{equation}
    are nonnegative. Since we have that
    \[
        (\rho_{j-1}^n- \rho_{\min}) \geq 0, \quad \text{and } (\rho_{j}^n- \rho_{\min}) \geq 0,
    \]
    it suffices to show that
    \[
        \frac{\partial \mathcal{H}}{\partial \rho_{j-1}^n} ( \tilde{\rho}_{j-1}^n, \tilde{\rho}_{j}^n, \rho_{j+1}^n, \dots, \rho_{j+m}^n)\geq 0 \quad \text{and }
        \frac{\partial \mathcal{H}}{\partial \rho_{j}^n} ( \tilde{\rho}_{j-1}^n, \tilde{\rho}_{j}^n, \rho_{j+1}^n, \dots, \rho_{j+m}^n)\geq 0.
    \]

    Since the derivative of $\mathcal{H}$ with respect to $\rho_{j-1}^n$ is given by
    \[
        \frac{\partial \mathcal{H}}{\partial \rho_{j-1}^n}(\rho_{j-1}^n, \rho_{j}^n, \rho_{j+1}^n, \dots, \rho_{j+m}^n)
        =\lambda(1-q_j^n),
    \]
    we obtain
    \begin{align*}
        \frac{\partial \mathcal{H}}{\partial \rho_{j-1}^n} ( \tilde{\rho}_{j-1}^n, \tilde{\rho}_{j}^n, \rho_{j+1}^n, \dots, \rho_{j+m}^n) = \lambda(1- \tilde{q}_j^n),
    \end{align*}
    where
    \[
        \tilde{q}_j^n := w_0 \tilde{\rho}_{j}^n+ \sum_{k=1}^{m-1} w_k \rho_{j+k}^n.
    \]
    By the induction hypothesis, we have
    \[
        0 \leq \rho_{\min} \leq  \rho_{j+k}^{n} \leq \rho_{\max} \leq 1 \qquad \text{ for } k = -1, 0,1,\dots,m.
    \] So, by a similar argument as \eqref{eq:q_j_equality}, we have $0 \leq \tilde{q}_j^n \leq 1$. Therefore, we obtain
    \[\frac{\partial \mathcal{H}}{\partial \rho_{j-1}^n} ( \tilde{\rho}_{j-1}^n, \tilde{\rho}_{j}^n, \rho_{j+1}^n, \dots, \rho_{j+m}^n) = \lambda(1- \tilde{q}_j^n) \geq 0.\]
    So, \eqref{eq:H_1_first_term} is nonnegative.

    It remains to show that \eqref{eq:H_1_second_term} is nonnegative. The derivative of $\mathcal{H}$ with respect to $\rho_{j}^n$ is given by
    \[
        \frac{\partial \mathcal{H}}{\partial \rho_{j}^n}(\rho_{j-1}^n, \rho_{j}^n, \rho_{j+1}^n, \dots, \rho_{j+m}^n)
        = 1-\lambda\left[(1-q_{j+1}^n)+w_0\rho_{j-1}^n\right].
    \]
    Then, we have
    \[
        \frac{\partial \mathcal{H}}{\partial \rho_{j}^n} ( \tilde{\rho}_{j-1}^n, \tilde{\rho}_{j}^n, \rho_{j+1}^n, \dots, \rho_{j+m}^n) =
        1-\lambda\left[(1-q_{j+1}^n)+w_0 \tilde{\rho}_{j-1}^n\right].\]
    Since $\lambda < \frac{1}{2}$ by Assumption~\ref{assm:assm_4}, and since $0\leq q_{j+1}^n \leq 1$ and $0 \leq w_0 \tilde{\rho}_{j-1}^n \leq 1$, we obtain
    \[1-\lambda\left[(1-q_{j+1}^n)+w_0 \tilde{\rho}_{j-1}^n\right]\geq 0.\]
    Thus, \eqref{eq:H_1_second_term} is nonnegative as well, and we conclude that $\Delta \mathcal{H}_1 \geq 0$.

    Next we show that $\Delta \mathcal{H}_2 \geq 0$. By the mean value theorem,
    there exist intermediate values
    \[ 0 \leq \rho_{\min} \leq \tilde{\rho}_{j+k}^n \leq \rho_{\max} \leq 1 \qquad \text{for } k= 1,2,\dots, m,\]
    such that
    \begin{align*}
        \Delta \mathcal{H}_2 & = \mathcal{H}( \rho_{\min}, \rho_{\min},  \rho_{j+1}^n,\dots,  \rho_{j+m}^n) - \mathcal{H}(\rho_{\min}, \rho_{\min}, \rho_{\min},\dots, \rho_{\min})                          \\
                             & = \sum_{k=1}^{m} \frac{\partial \mathcal{H}}{\partial \rho_{j+k}^n}\bigl(\rho_{\min},\rho_{\min},\tilde\rho_{j+1}^n,\dots,\tilde\rho_{j+m}^n\bigr)(\rho_{j+k}^n-\rho_{\min}).
    \end{align*}
    In order to have $\Delta \mathcal{H}_2 \geq 0$, we must show that
    \[
        \frac{\partial \mathcal{H}}{\partial \rho_{j+k}^n}
        \bigl(
        \rho_{\min},\rho_{\min},
        \tilde\rho_{j+1}^n,\dots,\tilde\rho_{j+m}^n
        \bigr)
        (\rho_{j+k}^n-\rho_{\min}) \geq 0 \qquad \text{ for } k= 1,2, \dots, m.
    \]
    Since
    \[
        (\rho_{j+k}^n-\rho_{\min}) \geq 0 \qquad \text{ for } k= 1,2, \dots, m,
    \]
    it suffices to show that
    \[
        \frac{\partial \mathcal{H}}{\partial \rho_{j+k}^n}
        \bigl(
        \rho_{\min},\rho_{\min},
        \tilde\rho_{j+1}^n,\dots,\tilde\rho_{j+m}^n
        \bigr)
        \geq 0 \qquad \text{ for } k= 1,2, \dots, m.
    \]

    For $k=m$, using the convention $w_m=0$, the derivative of $\mathcal{H}$ with respect to $\rho_{j+m}^n$ is given by
    \begin{align*}
        \frac{\partial \mathcal{H}}{\partial \rho_{j+m}^n}(\rho_{j-1}^n, \rho_{j}^n, \rho_{j+1}^n, \dots, \rho_{j+m}^n) & =\lambda\left(w_{m-1}\rho_j^n-w_m\rho_{j-1}^n\right) \\
                                                                                                                        & = \lambda w_{m-1}\rho_j^n
    \end{align*}
    Therefore,
    \begin{align*}
        \frac{\partial \mathcal{H}}{\partial \rho_{j+m}^n}
        \bigl(
        \rho_{\min},\rho_{\min},
        \tilde\rho_{j+1}^n,\dots,\tilde\rho_{j+m}^n
        \bigr)
         & = \lambda w_{m-1}\rho_{\min}                                                  \\
         & \geq \lambda cm^{-2}\rho_{\min} \qquad \text{by Assumption~\ref{assm:assm_3}} \\
         & \geq 0.
    \end{align*}
    For $k=1,\dots,m-1$, since we have
    \[ \frac{\partial \mathcal{H}}{\partial \rho_{j+k}^n}(\rho_{j-1}^n, \rho_{j}^n, \rho_{j+1}^n, \dots, \rho_{j+m}^n)
        =
        \lambda\left(w_{k-1}\rho_j^n-w_k\rho_{j-1}^n\right),\]
    we obtain
    \begin{align*}
        \frac{\partial \mathcal{H}}{\partial \rho_{j+k}^n}
        \bigl(
        \rho_{\min},\rho_{\min},
        \tilde{\rho}_{j+1}^n,\dots,\tilde\rho_{j+m}^n
        \bigr) & = \lambda \left(w_{k-1}\rho_{\min} - w_{k}\rho_{\min}\right) \\
               & = \lambda \rho_{\min} (w_{k-1}- w_{k})
    \end{align*}
    By Assumption~\ref{assm:assm_3}, we have $(w_{k-1}- w_{k}) \geq cm^{-2} > 0$. Therefore,
    \[
        \frac{\partial \mathcal{H}}{\partial \rho_{j+k}^n}
        \bigl(
        \rho_{\min},\rho_{\min},
        \tilde\rho_{j+1}^n,\dots,\tilde\rho_{j+m}^n
        \bigr)= \lambda \rho_{\min} \left(w_{k-1}- w_{k}\right) \geq 0 \qquad \text{for } k=1,\dots,m-1.
    \]
    Hence $\Delta \mathcal{H}_2 \geq 0$. We have shown that both $ \Delta \mathcal{H}_1 \geq 0 $ and $ \Delta \mathcal{H}_2 \geq 0$. Which implies
    \[
        \mathcal{H}(\rho_{j-1}^n,\rho_j^n,\rho_{j+1}^n,\dots,\rho_{j+m}^n)-\rho_{\min}
        =
        \Delta \mathcal{H}_1+\Delta \mathcal{H}_2 \geq 0 ,\]
    and therefore
    \[
        \mathcal{H}(\rho_{j-1}^n,\rho_j^n,\rho_{j+1}^n,\dots,\rho_{j+m}^n) \geq \rho_{\min}.
    \]
    The upper bound can be shown by similar argument. Thus, by induction,
    \[
        \rho_{\min} \leq \rho_j^n \leq \rho_{\max} \qquad \text{for all }  j \in \mathbb{Z} \quad \text{and } n \in \mathbb{N}.
    \]
\end{proof}

\subsection{One-sided Lipschitz Estimate}
Next, we prove the one-sided Lipschitz condition. The one-sided Lipschitz condition will play a crucial role in showing the monotonicity of the update function $\mathcal{H}$ given in \eqref{eq:mono_H}.
\begin{lemma}\label{lem:lemma_1}
    Suppose Assumptions~\ref{assm:assm_1}, \ref{assm:assm_3} and \ref{assm:assm_4} hold. In addition, suppose that
    \begin{enumerate}
        \item the initial data $\rho_0$ belongs to $L^\infty(\mathbb{R})$, has bounded total variation, and satisfies
              \[
                  0< \rho_{\min} \leq \rho_0(x) \le 1,  \qquad \text{for all } x \in \mathbb{R}
                  \qquad
              \]
              and
              \[
                  -\frac{\rho_0(y)-\rho_0(x)}{y-x} \le L
                  \qquad \text{for all } x \ne y \in \mathbb{R}
              \]
              for some constant $L>0$.

        \item \[
                  c\rho_{\min}\geq 2L\omega(0)\epsilon, \qquad \text{where $c$ is as in Assumption~\ref{assm:assm_3}}.
              \]
        \item
              \[
                  \epsilon= m \Delta x  \qquad \text{where $m \geq 1$ donets the number of cells}.
              \]
    \end{enumerate}
    Let $\{\rho_j^n\}_{j \in \mathbb{Z}}^{ n \in \mathbb{N}}$ be the numerical solution computed by \eqref{eq:mono_H}.
    Then, the numerical differences
    \begin{align*}
        r_j^n = \rho_{j+1}^n - \rho_j^n, \qquad j \in \mathbb{Z}, n \in \mathbb{N},
    \end{align*}
    satisfy the one-sided Lipschitz condition
    \begin{equation}\label{eq:one_sided_condition}
        r_j^n \geq -L\Delta x,  \qquad j \in \mathbb{Z}, n \in \mathbb{N}.
    \end{equation}
\end{lemma}
We highlight the main results in the proof rather rigorously. We begin by considering the structure of
\[
    r_j^{n+1} =  \rho_{j+1}^{n+1} - \rho_j^{n+1}.
\]
By definition in \eqref{eq:generic_H_N}, we have
\[
    \rho_{j+1}^{n+1} = \rho_{j+1}^{n} + \lambda\Bigl[ g(\rho_{j}^{n}, \rho_{j+1}^{n}, q_{j}^{n},q_{j+1}^{n})-g(\rho_{j+1}^{n}, \rho_{j+2}^{n}, q_{j+1}^{n},q_{j+2}^{n})\Bigr]
\]
and
\[
    \rho_{j}^{n+1} = \rho_{j}^{n} + \lambda\Bigl[ g(\rho_{j-1}^{n}, \rho_{j}^{n}, q_{j-1}^{n},q_{j}^{n})-g(\rho_{j}^{n}, \rho_{j+1}^{n}, q_{j}^{n},q_{j+1}^{n})\Bigr],
\]
where $g$ is the Godunov numerical flux.
Taking the difference gives us
\begin{equation*}
    \begin{split}
        r_j^{n+1} & = \rho_{j+1}^{n+1} - \rho_j^{n+1}                                                                                                                                                                                            \\
                  & = (\rho_{j+1}^n - \rho_j^n) + \lambda\Bigl[ 2g(\rho_{j}^{n}, \rho_{j+1}^{n}, q_{j}^{n},q_{j+1}^{n})-g(\rho_{j-1}^{n}, \rho_{j}^{n}, q_{j-1}^{n},q_{j}^{n})- g(\rho_{j+1}^{n}, \rho_{j+2}^{n}, q_{j+1}^{n},q_{j+2}^{n})\Bigr] \\
                  & =  r_j^n +  \lambda\Bigl[ 2g(\rho_{j}^{n}, \rho_{j+1}^{n}, q_{j}^{n},q_{j+1}^{n})-g(\rho_{j-1}^{n}, \rho_{j}^{n}, q_{j-1}^{n},q_{j}^{n})- g(\rho_{j+1}^{n}, \rho_{j+2}^{n}, q_{j+1}^{n},q_{j+2}^{n})\Bigr].
    \end{split}
\end{equation*}
As $g$ is quadratic, $r_j^{n+1}$ can be written as
\begin{equation}\label{eq:differences_God}
    r_j^{n+1}
    =
    \sum_{k=-1}^m c_{j,k}^n r_{j+k}^n
    +2\lambda(r_j^n)^2,
\end{equation}
where
\begin{align*}
    c_{j,-1}^n & = \lambda(1-q_j^n)                                           \\
    c_{j,0}^n  & = 1 - \lambda (1- q_j^n) + \lambda p_{j,0}^n -2\lambda r_j^n \\
    c_{j,k}^n  & = \lambda p_{j,k}^n, \qquad \text{for } k= 1,2,\dots,m,
\end{align*}
and
\[ p_{j,k}^n =  w_{k-1}\rho_{j+1}^n - w_k\rho_j^n + w_k r_j^n \qquad \text{with convention } w_{-1}= w_m = 0.\]

Now we prove the Lemma by induction
\begin{proof}[Proof of Lemma~\ref{lem:lemma_1}]
    Following \cite{huangAsymptoticCompatibilityClass2024}, we do the analysis for $\Delta x = \epsilon m^{-1}$. The one-sided Lipschitz condition on $\rho_0$ gives $r_j^0 \geq - L\Delta x$ for all $j\in \mathbb{Z}$. Assume now that \eqref{eq:one_sided_condition} holds for some $n \in \mathbb{N}$, that is,
    \[
        r_j^n \geq - L\Delta x,  \qquad j \in \mathbb{Z}.
    \]
    We show also \eqref{eq:one_sided_condition} holds for $n+1$. It suffices to show that $r_j^{n+1}$ is a convex combination of $r_{j-1}^n, r_{j}^n,\dots,r_{j+m}^n$ plus the nonnegative term $2\lambda(r_j^n)^2$, since then
    \[
        \inf\limits_{j \in \mathbb{Z}} r_j^{n+1} \geq \inf\limits_{j \in \mathbb{Z}} r_j^{n} \geq -L\Delta x,
    \]
    which gives the desired result.

    Since the term $2\lambda(r_j^n)^2$ in \eqref{eq:differences_God} is nonnegative and $\sum_{k=-1}^m c_{j,k}^n = 1$, it remains to show $c_{j,k}^n  \geq 0 $ for all  $k= -1,0,1,2,\dots,m$.

    Clearly, $c_{j,-1}^n \geq 0$ because we have shown that in \eqref{eq:First_argument}
    \begin{align*}
        c_{j,-1}^n & =\frac{\partial \mathcal{H}}{\partial \rho_{j-1}^n} = \lambda(1-q_j^n) \geq 0 \qquad \text{as } 0 \leq q_j^n \leq 1.
    \end{align*}
    For $k= 0$, we have
    \begin{align*}
        c_{j,0}^n & = 1 - \lambda (1- q_j^n) + \lambda p_{j,0}^n -2\lambda r_j^n                            \\
                  & = 1 - \lambda (1- q_j^n) + \lambda \Bigl(-w_0\rho_j^n + w_0 r_j^n\Bigr) -2\lambda r_j^n \\
                  & = 1- \lambda \Bigl[(1- q_j^n)+ w_0\rho_j^n  + (2-w_0)r_j^n\Bigr]
    \end{align*}
    In order to have  $c_{j,0}^n \geq 0$, it holds to show that $\lambda \Bigl[(1- q_j^n)+ w_0\rho_j^n  + (2-w_0)r_j^n\Bigr] \leq 1$. But by \eqref{eq:lambda_God} we have $\lambda < \frac{1}{2}$. Therefore, it is sufficient to show that
    \[
        (1- q_j^n)+ w_0\rho_j^n  + (2-w_0)r_j^n \leq 2.
    \]
    Note that all the terms $(1- q_j^n)$, $w_0\rho_j^n$ and $(2-w_0)$ are nonnegative since we have $ 0\leq q_j^n \leq 1$,  $0 \leq  w_0 \leq 1$ and  $0\leq \rho_j^n \leq 1$. We assume first that $r_j^n \leq 0$, then we have $(2-w_0)r_j^n \leq 0$ and
    \begin{align*}
        (1- q_j^n)+ w_0\rho_j^n  + (2-w_0)r_j^n & \leq (1- q_j^n)+ w_0\rho_j^n                                                                \\
                                                & \leq (1- q_j^n)+ \rho_j^n                                                                   \\
                                                & \leq (1-q_j^n) + \rho_{\max} \qquad \text{By maximum principle Theorem~\ref{the:theorem_1}} \\
                                                & \leq 2
    \end{align*}
    If $r_j^n >0$, then we have $(2-w_0)r_j^n >0$. In this case, the term
    \[
        (1- q_j^n)+ w_0\rho_j^n  + (2-w_0)r_j^n
    \]
    is nonnegative and we must show that
    \[
        (1- q_j^n)+ w_0\rho_j^n  + (2-w_0)r_j^n \leq 2,
    \]
    or equivalently
    \begin{equation}\label{eq:C0}
        \begin{split}
            (1- q_j^n)+ w_0\rho_j^n \leq 2 - (2-w_0)r_j^n
        \end{split}
    \end{equation}
    Huang and Du~\cite[p.~13]{huangAsymptoticCompatibilityClass2024}, claimed that this follows from Assumptions~\ref{assm:assm_1}--\ref{assm:assm_4} and the conditions in Lemma~\ref{lem:lemma_1}. But this does not seem as obvious as they claim. Because from the assumptions we can estimate the right hand side of \eqref{eq:C0} as
    \begin{align*}
        (1- q_j^n)+ w_0\rho_j^n & \leq (1- q_j^n)+ \rho_j^n                                                                   \\
                                & \leq (1-q_j^n) + \rho_{\max} \qquad \text{By maximum principle Theorem~\ref{the:theorem_1}} \\
                                & \leq 2,
    \end{align*}
    and the right hand side of \eqref{eq:C0} is strictly less than $2$, since the term $(2-w_0)r_j^n$ positive.
    Since we have not got enough time to investigate this part properly, we continue to prove the rest of the proof.

    For $ k= 1,2,\dots,m$, we have $c_{j,k}^n  = \lambda p_{j,k}^n$. So, it suffices to show that $p_{j,k}^n \geq 0$ for $ k= 1,2,\dots,m.$ We have
    \begin{align*}
        p_{j,k}^n & =  w_{k-1}\rho_{j+1}^n - w_k\rho_j^n + w_k r_j^n                       \\
                  & = (w_{k-1} - w_k)\rho_{j+1}^n + w_k(\rho_{j+1}^n-\rho_j^n) + w_k r_j^n \\
                  & = (w_{k-1} - w_k)\rho_{j+1}^n +  w_k r_j^n +  w_k r_j^n                \\
                  & = (w_{k-1} - w_k)\rho_{j+1}^n + 2w_k r_j^n.
    \end{align*}
    The terms $(w_{k-1} - w_k)$, $\rho_{j+1}^n $ and $2w_k$ are nonnegative since we have

    \[
        (w_{k-1} - w_k)\geq cm^{-2}, \quad w_k \geq 0 \qquad \text{by Assumptions~\ref{assm:assm_2}--\ref{assm:assm_3}},
    \]
    and
    \[
        \rho_{j+1}^n \geq \rho_{\min} > 0 \qquad \text{by maximum principle, Theorem~\ref{the:theorem_1}}.
    \]
    So, if $r_j^n \geq 0$, then we have $2w_k r_j^n \geq 0$ and
    \begin{align*}
        (w_{k-1} - w_k)\rho_{j+1}^n + 2w_k r_j^n & \geq cm^{-2}\rho_{\min} + 2w_k r_j^n \\
                                                 & \geq cm^{-2}\rho_{\min} \geq 0       \\
    \end{align*}
    If $r_j^n < 0$, then we have
    \begin{equation}\label{eq:w_k_equality}
        \begin{split}
            w_k & \leq \omega^{\epsilon}(k\Delta x) \Delta x \qquad \text{by Assumption~\ref{assm:assm_3}}                                     \\
                & = \frac{1}{\epsilon} \omega\!\left(\frac{k\Delta x}{\epsilon}\right) \Delta x  \qquad \text{by Assumption~\ref{assm:assm_1}} \\
                & \leq \omega(0) \frac{\Delta x}{\epsilon} \qquad \text{since $\omega$ is decreasing on $[0,1]$}                               \\
                & = \omega(0)m^{-1} \qquad \text{since $\epsilon = m \Delta x$}.
        \end{split}
    \end{equation}
    \
    So $w_k \leq \omega(0)m^{-1}$ implies  $2w_kr_j^n  \geq 2\omega(0)m^{-1} r_j^n $.
    In this case, we obtain
    \begin{align*}
        (w_{k-1} - w_k)\rho_{j+1}^n + 2w_k r_j^n & \geq cm^{-2}\rho_{\min} + 2\omega(0)m^{-1}r_j^n                                                                   \\
                                                 & \geq cm^{-2}\rho_{\min} - 2\omega(0)m^{-1}L\Delta x  \quad \text{by induction hypothesis}                         \\
                                                 & = cm^{-2}\rho_{\min} - 2\omega(0)m^{-1} L \epsilon m^{-1}   \quad \text{since $\Delta x = \epsilon m^{-1}$}       \\
                                                 & = m^{-2}( c\rho_{\min} - 2\omega(0)L \epsilon)\geq 0  \quad \text{since $c\rho_{\min} \geq 2\omega(0)L \epsilon$}
    \end{align*}
    Hence $p_{j,k}^n \geq 0$, and we conclude $c_{j,k}^n \geq 0$.
\end{proof}
\subsection{Monotonicity of the update function $\mathcal{H}$}
Having Lemma~\ref{lem:lemma_1} at hand, we can finally deduce that the update function $\mathcal{H}$ given by \eqref{eq:mono_H} is nondecreasing in each of its arguments.
\begin{proposition}\label{prop:mono_of_H}
    Under conditions of the Lemma~\ref{lem:lemma_1}, the update function $\mathcal{H}$ given by \eqref{eq:mono_H} is monotone.
\end{proposition}
\begin{proof}
    In Subsection~\ref{subsec:mono}, we have shown that $\mathcal{H}$ is nondecreasing with respect to $\rho_{j-1}, \rho_{j}$ and $\rho_{j+m}$. Now consider $k= 1,2,\dots,m-1$. We have
    \begin{align*}
        \frac{\partial \mathcal{H}}{\partial \rho_{j+k}^n} & = \lambda\left( w_{k-1}\rho_j^n -w_k\rho_{j-1}^n\right)                      \\
                                                           & =\lambda \Bigl[(w_{k-1}- w_k) \rho_j^n + w_k(\rho_j^n - \rho_{j-1}^n )\Bigr] \\
                                                           & = \lambda \Bigl[(w_{k-1}- w_k) \rho_j^n + w_k r_{j-1}^n\Bigr].
    \end{align*}
    Since $\lambda >0$, it suffices to show that $(w_{k-1}- w_k) \rho_j^n + w_k r_{j-1}^n \geq 0$.

    If $r_{j-1}^n \geq 0 $, then we have
    \begin{align*}
        (w_{k-1}- w_k) \rho_j^n + w_k r_{j-1}^n & \geq cm^{-2}\rho_{\min} + w_k r_{j-1}^n \\
                                                & \geq cm^{-2}\rho_{\min}  \geq 0.
    \end{align*}

    If $r_{j-1}^n < 0$, then we have $w_k \leq \omega(0)m^{-1}$ by \eqref{eq:w_k_equality}. This implies  $w_k r_{j-1}^n  \geq \omega(0)m^{-1} r_{j-1}^n  $. Therefore, we obtain
    \begin{align*}
        (w_{k-1}- w_k) \rho_j^n + w_k r_{j-1}^n & \geq cm^{-2}\rho_{\min} + \omega(0)m^{-1}r_{j-1}^n                                                                                               \\
                                                & \geq cm^{-2}\rho_{\min} -\omega(0)m^{-1}L\Delta x \qquad \text{by Lemma~\ref{lem:lemma_1}}\                                                      \\
                                                & = cm^{-2}\rho_{\min} -\omega(0)m^{-1}L\epsilon m^{-1} \qquad \text{since $\Delta x = \epsilon m^{-1}$ }                                          \\
                                                & = m^{-2}\Bigl(c\rho_{\min}- \omega(0)L\epsilon\Bigr) \geq 0 \qquad \text{ since $c\rho_{\min} \geq 2\omega(0)L\epsilon \geq \omega(0)L\epsilon$}
    \end{align*}
    Therefore, $\mathcal{H}$ is nondecreasing with respect to all its arguments, which means  $\mathcal{H}$ is monotone.
\end{proof}
\subsection*{Summary}
The maximum principle, Theorem~\ref{the:theorem_1}, gives $\mathbf{L^{\infty}}$ bounds. The total variation estimates are obtained from the monotonicity of $\mathcal{H}$. The Oleinike's entropy condition can be obtained from a careful analysis on the one sided Lipschitz condition, see \cite[LEMMA~4.3]{huangAsymptoticCompatibilityClass2024}. Thus, the numerical solutions computed by \eqref{eq:mono_H} is uniformly bounded in $\mathbf{BV}_{loc}(\mathbb{R} \times \mathbb{R}_{+})$. So, Helly's theorem gives precompactness in $\mathbf{L}_{loc}^{1}$. Then, there exist a convergent subsequence in $\mathbf{L}_{loc}^{1}(\mathbb{R} \times \mathbb{R}_{+})$ to a limit function $\rho^{*}$ by Ascoli's theorem. Finally we can apply the Lax--Wendroff theorem to prove that the limit function $\rho^{*}$ is the entropy solution of the local model, then by uniqueness of the entropy solution the whole sequence converge to the same limit function. That is, when we have
\[
    0 < \epsilon \leq \frac{c\rho_{\min}}{2L\omega(0)},
\]
then the numerical solution converges to the unique entropy solution $u$ of the local model \eqref{eq:local model} as both $\Delta x \rightarrow 0$ and $\epsilon \rightarrow 0$ simultaneously.
